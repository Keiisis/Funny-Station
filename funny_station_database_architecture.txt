funny_station_database_architecture
===================================

-- ══════════════════════════════════════════════════════════════
-- FUNNY STATION : DATABASE ARCHITECTURE (Universal Console)
-- ══════════════════════════════════════════════════════════════

-- 1. Types de Runtimes supportés
DO $$ BEGIN
    CREATE TYPE game_runtime AS ENUM ('html5', 'wasm_bridge', 'pyodide', 'lua_fengari', 'cheerpj');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- 2. Table des Jeux (Métadonnées et Configuration)
CREATE TABLE IF NOT EXISTS public.fs_games (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title           TEXT NOT NULL,
    slug            TEXT UNIQUE NOT NULL,
    description     TEXT,
    author_id       UUID REFERENCES auth.users(id),
    runtime         game_runtime NOT NULL DEFAULT 'html5',
    version         TEXT DEFAULT '1.0.0',
    
    -- Configuration technique
    entry_point     TEXT NOT NULL,        -- ex: 'index.html', 'main.py', 'game.wasm'
    dependencies    JSONB DEFAULT '{}',    -- ex: {"python_libs": ["numpy"], "wasm_memory": "256MB"}
    
    -- Assets visuels
    icon_url        TEXT,
    hero_banner_url TEXT,
    bg_music_url    TEXT,
    
    -- Statistiques
    play_count      BIGINT DEFAULT 0,
    rating          NUMERIC(3,2) DEFAULT 0,
    
    created_at      TIMESTAMPTZ DEFAULT now()
);

-- 3. Table des Trophées (Gamification)
CREATE TABLE IF NOT EXISTS public.fs_trophies (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    game_id         UUID REFERENCES public.fs_games(id) ON DELETE CASCADE,
    title           TEXT NOT NULL,
    description     TEXT,
    rarity          TEXT CHECK (rarity IN ('bronze', 'silver', 'gold', 'platinum')),
    hidden          BOOLEAN DEFAULT false,
    points          INTEGER DEFAULT 10
);

-- 4. Progression & Cloud Saves (VFS Sync)
CREATE TABLE IF NOT EXISTS public.fs_user_progress (
    user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    game_id         UUID REFERENCES public.fs_games(id) ON DELETE CASCADE,
    save_data       JSONB DEFAULT '{}',    -- Données de sauvegarde synchronisées
    last_played     TIMESTAMPTZ DEFAULT now(),
    total_time_mins INTEGER DEFAULT 0,
    PRIMARY KEY (user_id, game_id)
);

-- 5. Trophées débloqués
CREATE TABLE IF NOT EXISTS public.fs_unlocked_trophies (
    user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    trophy_id       UUID REFERENCES public.fs_trophies(id) ON DELETE CASCADE,
    unlocked_at     TIMESTAMPTZ DEFAULT now(),
    PRIMARY KEY (user_id, trophy_id)
);

-- 6. Index pour la performance (UX PS5 fluide)
CREATE INDEX IF NOT EXISTS idx_fs_games_runtime ON public.fs_games(runtime);
CREATE INDEX IF NOT EXISTS idx_fs_progress_user ON public.fs_user_progress(user_id);

-- 7. RLS (Security) - Les utilisateurs ne peuvent modifier que leurs propres sauvegardes
ALTER TABLE public.fs_user_progress ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage their own saves" ON public.fs_user_progress
    FOR ALL USING (auth.uid() = user_id);
